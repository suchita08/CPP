#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
#define n 1000000
bool primes[1000001];

//function to check if numbers are prime or not using Sieve of Eratosthenes 
//Time complexity : O(n*log(log(n)))


void sieve()
{
    ll i;
    for(i=0;i<n;i++)
    {
        primes[i]=true;
    }
    primes[0]=false;
    primes[1]=false;
    for(i=2;i*i<=n;i++)
    {
        if(primes[i])
        {
            for(ll j=i*i;j<n;j+=i)
            primes[j]=false;
        }
    }
}
int main()
{
        ios_base::sync_with_stdio(false);
        cin.tie(NULL);
        sieve();
        int t;
        cin>>t;
        while(t--)
        {
            ll num;
            cin>>num;
            cout<<primes[num]<<"\n";
        }
    return 0;
    
}
